-	script	::Sailor_izlude	-1,{
	mes "[船員]";
	mes "各位, 注意了!";
	mes "這艘擁有超高航速的船早已准備好隨時出發,";
	mes "不過它不能保証大家的安全就是了!";
	mes "來吧, 我們走!";
	mes "快點啊, 別磨磨蹭蹭的!";
	next;
	switch(select("柏伊亞嵐島 -> 150 Zeny.:艾爾貝塔碼頭  -> 500 Zeny.:取消.")) {
	case 1:
		if (Zeny < 150) {
			mes "[船員]";
			mes "150 Zeny!";
			mes "僅僅需要 150 Zeny 即可搭乘!";
			close;
		}
		set Zeny, Zeny - 150;
		warp "izlu2dun",107,50;
		end;
	case 2:
		if (Zeny < 500) {
			mes "[船員]";
			mes "500 Zeny!";
			mes "僅僅需要 500 Zeny 即可搭乘!";
			close;
		}
		set Zeny, Zeny - 500;
		warp "alberta",188,169;
		end;
	case 3:
		close;
	}
}
